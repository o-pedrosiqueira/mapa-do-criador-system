[Code]
procedure RunStep(Msg: String; Cmd: String; StepNum: Integer; Total: Integer);
var
  ResultCode: Integer;
begin
  WizardForm.StatusLabel.Caption := Msg;
  WizardForm.ProgressGauge.Position := ((StepNum - 1) * WizardForm.ProgressGauge.Max) div Total;
  Exec(ExpandConstant('{sys}\WindowsPowerShell\v1.0\powershell.exe'),
    '-ExecutionPolicy Bypass -NoProfile -WindowStyle Hidden -Command "' + Cmd + '"',
    '', SW_HIDE, ewWaitUntilTerminated, ResultCode);
  WizardForm.ProgressGauge.Position := (StepNum * WizardForm.ProgressGauge.Max) div Total;
end;

procedure CurStepChanged(CurStep: TSetupStep);
var
  PathRefresh: String;
  InstallDir: String;
begin
  if CurStep = ssInstall then
  begin
    WizardForm.ProgressGauge.Max := 100;

    PathRefresh := '$env:Path=[System.Environment]::GetEnvironmentVariable(''Path'',''Machine'')+'';''+[System.Environment]::GetEnvironmentVariable(''Path'',''User''); ';
    InstallDir := '$p=$env:USERPROFILE+''\Documents\mapa-do-criador''; ';

    RunStep('Instalando Python 3... (passo 1 de 7)',
      'winget install Python.Python.3.12 --silent --accept-package-agreements --accept-source-agreements',
      1, 7);

    RunStep('Instalando Git... (passo 2 de 7)',
      'winget install Git.Git --silent --accept-package-agreements --accept-source-agreements',
      2, 7);

    RunStep('Instalando Node.js LTS... (passo 3 de 7)',
      'winget install OpenJS.NodeJS.LTS --silent --accept-package-agreements --accept-source-agreements',
      3, 7);

    RunStep('Instalando Claude da Anthropic... (passo 4 de 7)',
      'winget install Anthropic.Claude --silent --accept-package-agreements --accept-source-agreements',
      4, 7);

    RunStep('Baixando o Mapa do Criador... (passo 5 de 7)',
      PathRefresh +
      InstallDir +
      'if (Test-Path ($p+''\.git'')) { Set-Location $p; git pull } else { git clone -b main https://github.com/pedrosiqueira/mapa-do-criador.git $p }',
      5, 7);

    RunStep('Instalando dependencias do painel... (passo 6 de 7)',
      PathRefresh +
      InstallDir +
      'Set-Location $p; npm install',
      6, 7);

    RunStep('Criando atalho na area de trabalho... (passo 7 de 7)',
      InstallDir +
      'Set-Content -Path ($p+''\iniciar-mapa.bat'') -Value (''@echo off''+[char]13+[char]10+''npm start'') -Encoding ASCII; ' +
      '$sh=New-Object -ComObject WScript.Shell; ' +
      '$sc=$sh.CreateShortcut($env:USERPROFILE+''\Desktop\Mapa do Criador.lnk''); ' +
      '$sc.TargetPath=$p+''\iniciar-mapa.bat''; ' +
      '$sc.WorkingDirectory=$p; ' +
      '$sc.WindowStyle=1; ' +
      '$sc.Save()',
      7, 7);
  end;
end;
